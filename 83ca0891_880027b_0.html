1. Worker实现了Runnable接口，实际上在线程中运行的是Worker对象，Worker对象从阻塞队列中不断取任务进行执行；

private final class Worker
    extends AbstractQueuedSynchronizer
    implements Runnable

 

2. 主要属性

private static final long serialVersionUID = 6138294804551838833L;
// 运行worker的线程
final Thread thread;
// 创建worker对象时，设置的第一个任务
Runnable firstTask;
// 统计该worker完成的任务数
volatile long completedTasks;

 

3. 构造器

Worker(Runnable firstTask) {
    // 设置初始的worker状态
    setState(-1);
    // 第一个任务
    this.firstTask = firstTask;
    // 创建线程，传入this，线程即执行 Worker#run 方法
    this.thread = getThreadFactory().newThread(this);
}

 

4. run方法：调用ThreadPoolExecutor的runWorker方法；

public void run() {
    runWorker(this);
}

 

5. runWorker：线程复用原理，重复地在阻塞队列中获取任务，然后执行；

final void runWorker(Worker w) {
    // 获取当前线程，即Worker中创建的线程，等价于w.thread
    Thread wt = Thread.currentThread();
    // 获取第一个任务
    Runnable task = w.firstTask;
   
    // 将w的firstTask置为null，因为接下来就是执行firstTask了
    w.firstTask = null;
    w.unlock();
   
    // worker线程是没有获取任务，超时结束，
    // 还是因为线程执行过程中，抛出异常,导致worker声明周期结束
    // completedAbruptly：突然停止
    boolean completedAbruptly = true;
    try {
        // 线程复用原理，重复在阻塞队列中获取任务，并执行
        // task != null
        //      或
        // getTask()是从阻塞队列中获取任务
        while (task != null || (task = getTask()) != null) {
            w.lock();

            /*
             线程>=STOP态 || (线程已中断 &&  线程>=STOP态）
                并
              线程未中断

            这里即当线程池>=STOP态时，会阻止线程运行
            Thread.interrupted()：会清除中断状态，第一次返回true，第二次会返回false，所以在后面的&&条件为判断线程是否未被中断
            wt.isInterrupted()：不会清除中断状态
              */
            if ((runStateAtLeast(ctl.get(), STOP) || (Thread.interrupted() && runStateAtLeast(ctl.get(), STOP)))
                    && !wt.isInterrupted())
                // 线程中断
                wt.interrupt();

            try {
                // 执行前回调函数
                beforeExecute(wt, task);
                Throwable thrown = null;
                try {
                    // 执行任务，直接调用run方法
                    task.run();
                } catch (RuntimeException x) {
                    thrown = x; throw x;
                } catch (Error x) {
                    thrown = x; throw x;
                } catch (Throwable x) {
                    thrown = x; throw new Error(x);
                } finally {
                    // 执行后回调函数
                    afterExecute(task, thrown);
                }
            } finally {
                task = null;
                // 完成任务数+1
                w.completedTasks++;
                w.unlock();
            }
        }
        // 达到这里，说明是因为没有获取到任务超时结束，不是被抛出异常终止的，所以completedAbruptly = false
        // 并不是突然停止
        completedAbruptly = false;
    } finally {
        // completedAbruptly == true，则worker数-1，否则不减，因为在getTask中已经-1了
        // 但一定会将worker从workers列表中移除
        processWorkerExit(w, completedAbruptly);
    }
}

 

6. getTask：从阻塞队列中获取任务；

private Runnable getTask() {
    // 是否超时标记
    boolean timedOut = false;

    // cas获取任务
    for (;;) {
        int c = ctl.get();
        int rs = runStateOf(c);

        /*
            线程池 >= SHUTDOWN态
                并
            线程池 >= STOP态 || 阻塞队列为空
         */
        if (rs >= SHUTDOWN && (rs >= STOP || workQueue.isEmpty())) {
            // 线程数-1，不获取任务
            decrementWorkerCount();
            return null;
        }

        // 获取worker线程数
        int wc = workerCountOf(c);

        // allowCoreThreadTimeOut：核心线程数是否允许超时消耗
        // wc > corePoolSize：当前worker线程数是否大于核心线程数
        // timed：是否允许超时
        //      true：在阻塞队列获取元素时，设置超时时间，指定时间还没有元素，则返回
        //      false：在阻塞队列获取元素时，一直阻塞，直到有元素为止
        boolean timed = allowCoreThreadTimeOut || wc > corePoolSize;

        /*
                worker线程数> maximumPoolSize || (允许超时 && 已经超时)
                    并
                wc > 1 || 阻塞队列为空
          */
        if ((wc > maximumPoolSize || (timed && timedOut))
            && (wc > 1 || workQueue.isEmpty())) {
            // worker线程数-1，超时不再获取任务
            if (compareAndDecrementWorkerCount(c))
                return null;
            // cas worker线程数-1失败，重试
            continue;
        }

        try {
            // 超时获取任务：workQueue.poll(keepAliveTime, TimeUnit.NANOSECONDS)
            // 还是阻塞，直到获取为止：workQueue.take()
            Runnable r = timed ?
                workQueue.poll(keepAliveTime, TimeUnit.NANOSECONDS) :
                workQueue.take();

            // 获取到任务，返回
            if (r != null)
                return r;

            // 达到这里，说明已经超时了
            timedOut = true;
        } catch (InterruptedException retry) {
            timedOut = false;
        }
    }
}
