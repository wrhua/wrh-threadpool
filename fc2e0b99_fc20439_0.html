1. transfer方法

E transfer(E e, boolean timed, long nanos) {
    QNode s = null;
    /**
     *  当前的节点类型
     *  如果是put操作，那么e != null，表示是放数据的操作
     *  如果是take操作，那么 e == null，表示是取数据的操作
     */
    boolean isData = (e != null);

    // 死循环，纯cas操作，没有加锁处理，直到处理成功，或者超时、中断
    for (;;) {
        // 获取队列的头和尾
        QNode t = tail;
        QNode h = head;
        /**
         *   队列最少有一个空节点，t、h不可能等于null，
         *   等于null，则可能是并发操作，导致了可见性问题，此时continue，cas重试
         */
        if (t == null || h == null)
            continue;

        /**
         * h==t：表示队列为空，都指向一个空节点
         * t.isData == isData：队列中存放的节点类型和当前入队的节点类型相同
         */
        if (h == t || t.isData == isData) {
            // 获取队列最后一个节点的next，如果没有并发操作，tn应该等于null
            QNode tn = t.next;

            // 本来就是t 被赋值为tail，到这里，如果t != tail，则说明发生了并发操作，
            // continue，cas重试
            if (t != tail)
                continue;

            // 最后一个节点的next应该等于null，不等于null，则说明发生了并发操作
            if (tn != null) {
                // 尝试帮忙推动修改尾节点的指针，即将tail指向tn
                advanceTail(t, tn);
                // continue，cas重试
                continue;
            }

            // 不进行等待，因为节点类型相同，或者只有个空节点，那就是没有匹配的节点，此时直接返回null
            if (timed && nanos <= 0)
                return null;

            // 创建节点s
            if (s == null)
                s = new QNode(e, isData);

            // cas将s链接到队尾的最后一个节点上
            // 如果失败，则证明并发插入队尾，continue，cas重试
            if (!t.casNext(null, s))
                continue;

            // 将tail指向s节点
            advanceTail(t, s);

            /**
             * 自旋、阻塞等待，直到有节点和它互补，或是超时中断终止等待
             * 成功则返回互补节点的item
             * 失败则返回s节点本身
             */
            Object x = awaitFulfill(s, e, timed, nanos);

            // 如果返回的x == s，则证明是超时或是被中断，没有得到对应的互补元素
            if (x == s) {
                // 从队尾中清除s节点
                clean(t, s);
                return null;
            }

            //  s结点还没离开队列
            if (!s.isOffList()) {
                // 尝试将head指向s节点，即s节点变成新的空节点
                // 仅在head == tail时会发生这个操作
                advanceHead(t, s);

                // 返回值x != null，此时s.item == x
                // 这里为了将s节点置为空节点，就将s.item = s
                if (x != null)
                    s.item = s;

                // 等待线程为null，因为空节点不存在等待线程
                s.waiter = null;
            }

            // 返回一个元素，表示匹配成功
            return (x != null) ? (E)x : e;

        } else {
            // 到这里，说明t.isData != isData, 新节点和队列元素互补
            // 获取头节点
            QNode m = h.next;

            // t != tail：t 被赋值为 tail，此时t != tail ，则说明发生了并发操作
            // m == null：m 不应该为空，因为前面判断 h != t，说明队列有节点，如果m等于null，则说明发生了并发操作
            // h != head：h 赋值为 head，此时 h != head，则说明发生了并发操作
            if (t != tail || m == null || h != head)
                continue; // cas重试

            // 获取互补的节点的数据
            Object x = m.item;
            /**
             * isData == (x != null)：再次确认是互补节点，没有因为并发导致两个一样的节点进行匹配
             * x == m：节点没有被取消，如果被取消，x == m
             * !m.casItem(x, e)：值交换，将当前的e数据cas放入m失败，则说明存在线程已经消费了m节点
             */
            if (isData == (x != null) ||
                x == m ||
                !m.casItem(x, e)) {
                // 尝试推动head指向m，continue，cas重试
                advanceHead(h, m);
                continue;
            }

            // 到这里，说明和m节点互补成功了，此时队头节点出列，将head指向m
            advanceHead(h, m);
            // 唤醒在等待的waiter线程
            LockSupport.unpark(m.waiter);
            // 返回一个元素，表示匹配成功
            return (x != null) ? (E)x : e;
        }
    }
}

 

2. awaitFulfill方法

Object awaitFulfill(QNode s, E e, boolean timed, long nanos) {
    // 计算等待超时时间
    final long deadline = timed ? System.nanoTime() + nanos : 0L;

    // 当前线程
    Thread w = Thread.currentThread();

    // 自旋次数
    int spins = ((head.next == s) ?
                 (timed ? maxTimedSpins : maxUntimedSpins) : 0);

    // 循环等待，直到获取数据，或中断、超时为止
    for (;;) {
        // 如果被中断，则将s.item置为s自身
        if (w.isInterrupted())
            s.tryCancel(e);

        // 获取item值
        Object x = s.item;

        // x != e,则说明别的线程修改了s.item，或者是超时、中断，将s.item改成了s自身；
        // 返回x
        if (x != e)
            return x;

        // 如果设置了超时等待
        if (timed) {
            // 计算超时时间
            nanos = deadline - System.nanoTime();
            // 如果已经超过等待时间，则终止，将s.item改成了s自身；
            if (nanos <= 0L) {
                s.tryCancel(e);
                continue;
            }
        }

        // 如果自旋次数>0,自旋次数-1
        if (spins > 0)
            --spins;

        // 设置等待线程
        else if (s.waiter == null)
            s.waiter = w;

        // 如果没有设置超时等待，则一直阻塞，直到被唤醒
        else if (!timed)
            LockSupport.park(this);

        // 睡个nanos时间，直到被唤醒，或者是睡nanos时间后，自然醒
        else if (nanos > spinForTimeoutThreshold)
            LockSupport.parkNanos(this, nanos);
    }
}
